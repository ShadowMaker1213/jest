import {
  AxiosError,
  AxiosInstance,
  AxiosRequestConfig,
  AxiosResponse,
} from 'axios';
import assign from 'lodash/assign';
import { Dispatch } from 'react-redux';
import { MiddlewareAPI } from 'redux';

export const createSuccessActionType = (type: string) => {
  return `${type}_REQUEST_SUCCESS`;
};
 
export const createFailedActionType = (type: string) => {
  return `${type}_REQUEST_FAILED`;
};

export interface IRequestAction {
  type: string;
  payload: AxiosRequestConfig;
  meta: {
    request: true;
  };
}

export interface IAxiosResponse<TData> {
  data: TData;
  status: number;
  statusText: string;
  headers: any;
  config: AxiosRequestConfig;
}

export interface IRequestSuccessAction<TData> {
  type: string;
  payload: IAxiosResponse<TData>;
  meta: {
    previousAction: IRequestAction;
  };
}
 
export interface IRequestFailedAction {
  type: string;
  error: AxiosError;
  payload: any;
  meta: {
    previousAction: IRequestAction;
  };
}
 
export function createRequestAction<TReq>(type: string, reqMapper: (req: TReq) => AxiosRequestConfig) {
  const actionCreator = (req: TReq): IRequestAction => {
    return {
      type,
      payload: reqMapper(req),
      meta: {
        request: true,
      },
    };
  };
  const helpers = {
    toString: () => type,
    success: {
      toString: () => {
        return createSuccessActionType(type);
      },
    },
    failed: {
      toString: () => {
        return createFailedActionType(type);
      },
    },
  };
  assign(actionCreator, helpers);
  return actionCreator as typeof actionCreator & typeof helpers;
}
 
const isRequestAction = (action: IRequestAction) => {
  return action.meta && action.meta.request;
};
 
const createReqStartAction = (action: IRequestAction): any => {
  return {
    type: `${action.type}_REQUEST_START`,
    payload: action,
  };
};
 
const createSuccessAction = (action: IRequestAction, response: AxiosResponse): IRequestSuccessAction<any> => {
  return {
    ...action,
    type: createSuccessActionType(action.type),
    payload: response,
    meta: {
      previousAction: action,
    },
  };
};
 
const createFailedAction = (action: IRequestAction, error: AxiosError): IRequestFailedAction => {
  return {
    ...action,
    type: createFailedActionType(action.type),
    error,
    meta: {
      previousAction: action,
    },
  };
};
 
export const requestMiddleware: any = (client: AxiosInstance) => {
  return ({ dispatch }: MiddlewareAPI<any>) =>
    (next: Dispatch<any>) =>
      (action: IRequestAction) => {
        if (isRequestAction(action)) {
          dispatch(createReqStartAction(action));
          return client.request(action.payload)
            .then((response: AxiosResponse) => {
              return dispatch(createSuccessAction(action, response));
            })
            .catch((error: AxiosError) => {
              return dispatch(createFailedAction(action, error));
            });
        }
        return next(action);
      };
};
