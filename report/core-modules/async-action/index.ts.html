import { Dictionary } from 'lodash';
import {
  Dispatch,
  MiddlewareAPI,
} from 'redux';
import {
  createFailedActionType,
  createSuccessActionType,
} from '../request';
 
type TAsyncActionCallback = (previousAction: any, dispatch: Dispatch<any>) => void;
interface IAsyncAction {
  type: string;
  async: Dictionary<TAsyncActionCallback>;
}
 
export const asyncActionDispatcher = ({ dispatch }: MiddlewareAPI<any>) => {
  return (next: Dispatch<any>) => {
    return (action: IAsyncAction) => {
      const returnedValue = next(action);
 
      if (action.async && action.async[action.type]) {
        action.async[action.type](action, dispatch);
      }
 
      return returnedValue;
    };
  };
};

export class AsyncAction {
  static of(action: any) {
    return new AsyncAction(action);
  }
 
  asyncCallbacks: Dictionary<TAsyncActionCallback> = {};
 
  constructor(private action: any) {
    this.asyncCallbacks = action.async || {};
  }
 
  success(callback: TAsyncActionCallback) {
    return this.appendCallback(createSuccessActionType(this.action.type), callback);
  }
 
  failed(callback: TAsyncActionCallback) {
    return this.appendCallback(createFailedActionType(this.action.type), callback);
  }
 
  dispatchBy(dispatch: Dispatch<any>) {
    dispatch(this.toAction());
    return this;
  }
 
  toAction() {
    return {
      ...this.action,
      async: this.asyncCallbacks,
    };
  }
 
  appendCallback(type: string, callback: TAsyncActionCallback) {
    this.asyncCallbacks[type] = callback;
    return this;
  }
}
