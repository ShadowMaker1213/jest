import { AsyncAction } from 'core-modules/async-action';
import { Component } from 'react';
import {
  connect,
  DispatchProp,
} from 'react-redux';
import { Action } from 'redux-actions';

interface IWithEntitiesProps<TInnerProps> {
  groupName: string;
  actionCreator: () => Action<any>;
  children: (innerProps: TInnerProps) => JSX.Element | null;
  reducer: (state: any, action: Action<any>) => any;
  removeDataAfterDestroy?: boolean;
}
 
interface IWithEntitiesStateProps<TInnerProps> {
  state: TInnerProps;
}

type TWithEntitiesCoreProps<TInnerProps> =
  IWithEntitiesProps<TInnerProps>
  & DispatchProp<any>
  & IWithEntitiesStateProps<TInnerProps>;
 
class WithEntitiesCore<TInnerProps> extends Component<TWithEntitiesCoreProps<TInnerProps>, any> {
  static defaultProps = {
    removeDataAfterDestroy: true,
  };

  componentWillMount() {
    AsyncAction.of(this.props.actionCreator())
      .success((action: Action<any>) => {
        this.props.dispatch({
          type: 'UPDATE_DATA',
          meta: {
            groupName: this.props.groupName,
          },
          payload: {
            data: this.props.reducer({}, action),
          },
        });
      })
      .dispatchBy(this.props.dispatch);
  }
 
  componentWillUnmount() {
    if (this.props.removeDataAfterDestroy) {
      this.props.dispatch({
        type: 'DELETE_DATA',
        meta: {
          groupName: this.props.groupName,
        },
      });
    }
  }
 
  render() {
    return this.props.children(this.props.state);
  }
}
 
export const WithEntities = connect<IWithEntitiesStateProps<any>, any, IWithEntitiesProps<any>>((state: any, props) => {
  return {
    state: state.entities[props!.groupName!],
  };
})(WithEntitiesCore);
