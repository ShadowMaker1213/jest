import { css } from 'glamor';
import * as React from 'react';
import { Component } from 'react';

const basicStyle = {
  background: '#717171',
  boxShadow: '0 3px 0 #676767',
  lineHeight: '2.1em',
  textAlign: 'center',
  borderRadius: '.3125rem',
  fontSize: '1.3125em',
  color: '#fff',
  textTransform: 'uppercase',
  transition: 'all .1s',
  border: 'none',
  fontFamily: '"Francois One","Arial Narrow","Impact",Arial,Helvetica,sans-serif,sans-serif',
  ':focus': {
    outline: 0,
  },
};
 
const primaryStyle = {
  background: '#CD1719',
  boxShadow: '0 4px 0 #9f1213',
  padding: '0 2em',
};

interface IButtonProps {
  text: string;
  type: string;
  onClick: any;
}
 
export class Button extends Component<IButtonProps, any> {
  render() {
    const { type, text, onClick } = this.props;
    const styles = (type === 'primary') ? {
        ...basicStyle,
        ...primaryStyle,
      } : basicStyle;
    return (
      <button
        {...css(styles)}
        onClick={onClick}>
        {text}
      </button>
    );
  }
}
