import * as React from 'react';
import {
  Component,
  ReactNode,
} from 'react';

class AccordionGroup extends Component<{ children: ReactNode }, any> {
  accordionContent: any;
  state = {
    show: true,
  };
 
  toggle = () => {
    this.setState({
      show: !this.state.show,
    });
  }

  render() {
    return (
      <div>
        <div onClick={this.toggle}>AccordionHead</div>
        <div ref={(instance) => {
          this.accordionContent = instance;
        }} style={{ display: this.state.show ? 'block' : 'none' } }>{
          this.props.children
        }
        </div>
      </div>
    );
  }
}
 
export default AccordionGroup;
