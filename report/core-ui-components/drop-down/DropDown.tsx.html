import { css } from 'glamor';
import * as React from 'react';
import { Component } from 'react';
 
interface IList {
  key: string;
  value: any;
}
 
interface IDropDownProps {
  className?: string;
  name?: string;
  options: IList[];
}
 
const style = css({
  position: 'relative',
  border: '2px solid #dbdbdb',
 
  ' select': {
    appearance: 'none',
    background: '#fff',
    paddingLeft: '.5em',
    paddingRight: '1.8em',
    width: '100%',
    height: '2em',
    borderRadius: '0',
    border: '0',
  },
  ':after': {
    position: 'absolute',
    right: '1em',
    top: '.45em',
    borderStyle: 'solid',
    borderWidth: '0.25em 0.25em 0 0',
    display: 'inline-block',
    height: '0.45em',
    width: '0.45em',
    content: '""',
    transform: 'rotate(135deg)',
    verticalAlign: 'top',
    pointerEvents: 'none',
    color: '#949494',
  },
});
 
export class DropDown extends Component<IDropDownProps, any> {
  render() {
    const {options, ...others} = this.props;
    return (
      <div {...style} {...others}>
          <select>
            {options.map((opt, index) => {
                return (<option key={index} value={opt.key}>{opt.value}</option>);
            })}
          </select>
      </div>
    );
  }
}
