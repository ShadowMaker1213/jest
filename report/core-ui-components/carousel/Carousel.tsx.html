import { css } from 'glamor';
import first from 'lodash/first';
import * as React from 'react';
import { Component } from 'react';

interface IImgSize {
  big: string;
  small: string;
  medium?: string;
}

interface ICarouselProps {
  gap?: number;
  displayAreaWidth?: number;
  displayAreaHeight: number;
  imgWidth: number;
  imgResources: IImgSize[];
}
 
const suitableImgFromImgSize = (size: IImgSize | undefined) => {
  if (size === undefined) {
    return '';
  }
  return size.big || size.medium || size.small || '';
};

const navLink = css({
  height: '100%',
  width: '1.5rem',
  backgroundColor: 'gray',
  textIndent: '-9999px',
  position: 'absolute',
  top: 0,
  zIndex: 2,
  cursor: 'pointer',
  ':hover': {
    backgroundColor: '#ccc',
  },
});

export class Carousel extends Component<ICarouselProps, any> {
  static defaultProps = {
    gap: 10,
  };
 
  state = {
    currentPath: suitableImgFromImgSize(first(this.props.imgResources)!),
    currentIndex: 0,
  };

  private thumbsWrapper: HTMLDivElement;

  onPrevClick = () => {
    this.setScroll(false);
  }

  onNextClick = () => {
    this.setScroll(true);
  }
 
  setScroll(isPrev: boolean) {
    const { imgWidth, gap } = this.props;
    const imgCount: number = Math.floor(this.thumbsWrapper.clientWidth / (imgWidth + gap!));
    const currentScrollLeft = (this.thumbsWrapper as HTMLDivElement).scrollLeft;
    (this.thumbsWrapper as HTMLDivElement).scrollLeft = isPrev
      ? currentScrollLeft + (imgCount * imgWidth)
      : currentScrollLeft - (imgCount * imgWidth);
  }
 
  onImgClick(_: MouseEvent, imgSize: IImgSize, currentIndex: number) {
    this.setState({
      currentPath: suitableImgFromImgSize(imgSize),
      currentIndex,
    });
  }
 
  changeToNextPic = () => {
    const { currentIndex } = this.state;
    const { imgResources } = this.props;
    if (currentIndex < (imgResources.length - 1)) {
      this.setState({
        currentIndex: currentIndex + 1,
        currentPath: suitableImgFromImgSize(this.props.imgResources[currentIndex + 1]),
      });
    }
  }

  render() {
    const {
      imgResources,
      imgWidth,
      displayAreaHeight,
      gap,
      displayAreaWidth,
    } = this.props;
    return (
      <div>
        <div>
          <div {...css({
            maxWidth: '100%',
            maxHeight: '26rem',
            display: 'inline-block',
          })}
               onClick={this.changeToNextPic}
          >
            <img
              src={this.state.currentPath}
              alt=''
              {...css({
                maxWidth: '100%',
                maxHeight: `26rem`,
              })}/>
          </div>
        </div>
        <div {...css({
          position: 'relative',
          height: displayAreaHeight,
          maxWidth: displayAreaWidth ? displayAreaWidth : '100%',
        })}>
          <a {...navLink} {...css({ left: 0 })} onClick={this.onPrevClick}>Left</a>
          <a {...navLink} {...css({ right: 0 })} onClick={this.onNextClick}>Right</a>
          <div
            {...css({
              overflowX: 'auto',
              overflowY: 'hidden',
              margin: '0 1.7rem',
            })}
            ref={(thumbsWrapper: HTMLDivElement) => {
              this.thumbsWrapper = thumbsWrapper;
            }}
          >
            <div {...css({
              width: ((imgWidth + gap!) * imgResources.length) - gap!,
              height: displayAreaHeight,
              display: 'inline-flex',
            })}>
              {
                imgResources.map((imgSize: IImgSize, index: number) => {
                  return (
                    <a
                      key={index}
                      {...css({
                        cursor: 'pointer',
                        display: 'block',
                        marginLeft: index !== 0 && `${gap}`,
                        ':hover': { opacity: 0.7 },
                      })}
                      onClick={($event: any) => this.onImgClick($event, imgSize, index)}
                    >
                      <div {...css({
                        width: imgWidth,
                        minHeight: displayAreaHeight,
                        display: 'flex',
                        alignItems: 'center',
                        background: '#f2f2f2',
                        border: index === this.state.currentIndex ? '2px solid #CD1719' : 'none',
                      })}>
                        <img
                          src={imgSize.small}
                          alt=''
                          {...css({
                            maxHeight: displayAreaHeight,
                            maxWidth: imgWidth,
                            display: 'block',
                            margin: '0 auto',
                          })}
                        />
                      </div>
                    </a>
                  );
                })
              }
            </div>
          </div>
        </div>
      </div>
    );
  }
}
