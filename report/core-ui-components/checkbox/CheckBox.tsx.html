import { css } from 'glamor';
import * as React from 'react';
import { Component } from 'react';
// import { connect } from 'redux';

interface ICheckBoxProps {
    id?: string;
    disabled?: boolean;
    label: string;
}
 
const style = css({
  position: 'absolute',
  opacity: '0',
 
  ' + label': {
    position: 'relative',
    cursor: 'pointer',
    paddingLeft: '26px',
    minHeight: '19px',
    display: 'inline-block',
  },
 
  // Box.
  ' + label:before': {
    content: '""',
    position: 'absolute',
    // marginRight: '10px',
    border: '2px solid #dbdbdb',
    left: '0px',
    // display: 'inline-block',
    // verticalAlign: 'text-top',
    width: '19px',
    height: '19px',
    background: 'white',
  },
 
  // Box hover
  ':hover + label:before': {
    // background: '#f35429',
  },
 
  // Box focus
  ':focus + label:before': {
    borderColor: '#a8a8a8',
    boxShadow: '0 0 0 3px rgba(0, 0, 0, 0.12)',
  },
 
  // Box checked
  ':checked + label:before': {
    // background: '#f35429',
  },

  // Disabled state label.
  ':disabled + label': {
    color: '#b8b8b8',
    cursor: 'auto',
  },
 
  // Disabled box.
  ':disabled + label:before': {
    boxShadow: 'none',
    background: '#ddd',
  },
 
  // Checkmark. Could be replaced with an image
  ':checked + label:after': {
    content: '""',
    position: 'absolute',
    left: '5px',
    top: '7px',
    background: 'white',
    width: '13px',
    height: '10px',
    backgroundImage: 'url(https://www.gebrauchtwagen.at/file-vers1067/images/check.svg),none',
    backgroundPosition: '-75px 0',
    // boxShadow: '2px 0 0 white,4px 0 0 white,4px -2px 0 white,4px -4px 0 white,4px -6px 0 white,4px -8px 0 white',
  },
 
});
 
export class CheckBox extends Component<ICheckBoxProps, any> {
    render() {
      const {label, id, ...props} = this.props;
      return (
            <div {...css({display: 'inline-block'})}>
                <input
                    type='checkbox'
                    id={id}
                    {...style}
                    {...props}
                />
                <label htmlFor={id}>{label}</label>
            </div>
        );
    }
 
}
