import { css } from 'glamor';
import * as React from 'react';

const style = css({
  fontSize: '18px',
  fontWeight: 'normal',
  textDecoration: 'none',
  textAlign: 'center',
  wordSpacing: '0',
  backgroundColor: '#EDEDED',
  listStyle: 'none',
  padding: '8px 2px',
  width: '23.5%',
  height: '38px',
});

interface IInfoLabelProps {
  content: string;
  additionalContent?: string;
  color?: string;
}
 
export const InfoLabel = ({content, additionalContent, color}: IInfoLabelProps) => {
  const combinedStyle = color ? css(style, {color}) : style;
  return (
    <li {...combinedStyle}>
      {content}
      {
        additionalContent &&
        <span>{`(${additionalContent})`}</span>
      }
    </li>
  );
};
