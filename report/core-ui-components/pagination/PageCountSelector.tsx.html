import { css } from 'glamor';
import * as React from 'react';

const pageCountSelectorCSS = css({
  display: 'flex',
});

export interface IPageCountSelectorProps {
  prompt?: string;
  selectedPageSize: number;
  pageSizeOptions?: number[];
  onPageSizeChanged: (size: number) => void;
}
 
export class PageCountSelector extends React.Component<IPageCountSelectorProps, any> {
  state = {pageSize: this.props.selectedPageSize};

  handleChange = (e: any) => {
    const pageSize = e.target.value;
    this.setState({pageSize});
    this.props.onPageSizeChanged(pageSize);
  }

  render() {
    const {pageSizeOptions = [25, 50, 100]} = this.props;
    return (
      <div {...pageCountSelectorCSS} className='pages-count'>
        <label htmlFor=''>{this.props.prompt || 'per page'}</label>
        <div>
          <form action=''>
            <select value={this.state.pageSize} onChange={this.handleChange}>
              {
                pageSizeOptions.map((option, index) => (
                  <option value={option} key={index}>{option}</option>
                ))
              }
            </select>
          </form>
        </div>
      </div>
    );
  }
}
