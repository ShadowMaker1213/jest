import * as React from 'react';
import { Link } from 'react-router-dom';
import { IPageSelector } from './PageSelector';

export enum PageLinkType {
  normal,
  pre,
  next,
}

interface IPageLinkProps extends IPageSelector {
  page?: number;
  linkType: PageLinkType;
}

export const PageLink = ({page = 1, selectedPage, totalPages, pathForPage, linkType}: IPageLinkProps) => {
  switch (linkType) {
    case PageLinkType.pre:
      if (selectedPage === 1) {
        return <span>{'<'}</span>;
      }
      return (<Link to={pathForPage(selectedPage - 1)}>{'<'}</Link>);
    case PageLinkType.normal:
      return <Link to={pathForPage(page)}>{page === selectedPage ? `!${page}!` : `${page}`}</Link>;
    case PageLinkType.next:
      if (selectedPage === totalPages) {
        return <span>{'>'}</span>;
      }
      return (<Link to={pathForPage(selectedPage + 1)}>{'>'}</Link>);
  }
};
