import { css } from 'glamor';
import drop from 'lodash/drop';
import * as React from 'react';
import { PageLink, PageLinkType } from './PageLink';

const pageSelectorCSS = css({
  display: 'flex',
});

export interface IPageSelector {
  selectedPage: number;
  totalPages: number;
  pathForPage: (page: number) => string;
}

export const PageSelector = (props: IPageSelector) => {
  function generateArrayFromOneToN(n: number): number[] {
    return drop(Array.from(Array(n + 1).keys()), 1);
  }
 
  return (
    <div {...pageSelectorCSS} className='paging-wrapper'>
 
      <PageLink {...{...props, linkType: PageLinkType.pre}} />
      <div>
        {
          generateArrayFromOneToN(props.totalPages)
            .map((page, key) => (
              <PageLink {...{...props, linkType: PageLinkType.normal, page}} key={key} />
            ))
        }
      </div>
      <PageLink {...{...props, linkType: PageLinkType.next}} />
    </div>
  );
};
