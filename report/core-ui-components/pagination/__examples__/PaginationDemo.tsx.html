export class PaginationDemo extends Component<any, any> {
  static tsx = () => {
    return require(`!!raw-loader!./PaginationDemo.tsx`);
  }

  render() {
    const Container = (props: any) => {
      const search = new URLSearchParams(props.location.search);
      const searchPage = parseInt(search.get('page')!, 10) || 1;
      const searchPageSize = parseInt(search.get('size')!, 10) || 2;
 
      return (
        <Pagination
          items={fakeData.slice(((searchPage - 1) * searchPageSize), searchPage * searchPageSize)}
          totalPages={fakeData.length / searchPageSize}
          selectedPage={searchPage || 1}
          pathForPage={(page) => `/pagination?page=${page}&size=${searchPageSize}`}
          selectedPageSize={searchPageSize || 2}
          pageSizeOptions={[1, 2, 3]}
          onPageSizeChanged={(size) => {
            console.log(size);
          }}
        />
      );
    };
    return (
      <Route path='/pagination/' component={Container} />
    );
  }
