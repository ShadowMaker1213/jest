import * as React from 'react';
import { Action } from 'redux-actions';
import { WithEntities } from '../../../core-modules/entities/WithEntities';
import {
  ISearchResultListProps,
  SearchResultList,
} from '../../../core-ui-components/search-result-list/SearchResultList';
import { fetchSearchResultsList } from '../redux/actions';

export interface ISearchResultList extends ISearchResultListProps {
  pageData: any;
}
 
interface ISearchResultsPageProps {
  searchResultsList: ISearchResultList;
 
  fetchSearchList(): any;
}
 
const mapServerDataToState = (data: any): ISearchResultList => {
  return {
    items: data.listings.map((carInfo: any) => {
      const summary = carInfo.details.summary;
      const vendor = carInfo.details.vendor;
      return {
        img: {
          detailURL: `/car/${carInfo.id}`,
          imgURL: summary.images[0] && summary.images[0].medium || '',
          title: summary.title,
        },
        info: {
          title: summary.title,
          detailURL: `/car/${carInfo.id}`,
          location: `${vendor.address.zip} ${vendor.address.city}`,
          carType: '',
          quality: summary.vehicleOfferType.formatted,
          fuelType: summary.fuel.formatted,
          color: '#COLOR MISSING#',
          initialRegistration: summary.firstRegistrationDate.formatted,
          powerInKW: summary.power.kw.raw,
          mileage: summary.mileage.formatted,
          price: summary.price.formatted,
        },
      };
    }),
    pageData: data.metadata,
  };
};
 
type TChild = (innerProps: { images: string }) => JSX.Element | null;
 
const WithSearchResults = ({ children }: { children: TChild }) => (
  <WithEntities
    groupName='searchResults'
    actionCreator={() => fetchSearchResultsList(location.search)}
    reducer={(_: any, action: Action<any>) => {
      return {
        searchResultsList: mapServerDataToState(action.payload.data),
      };
    }}
  >
    {children}
  </WithEntities>
);
 
class SearchResultsPage extends React.Component<ISearchResultsPageProps, any> {
  render() {
    return (
      <WithSearchResults>
        {
          (innerProps: any) => innerProps ? <SearchResultList items={innerProps.searchResultsList.items}/> : null
        }
      </WithSearchResults>
    );
  }
}
 
export const SearchResults = SearchResultsPage;
