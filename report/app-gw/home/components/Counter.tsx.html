import { Button } from 'core-ui-components/button/Button';
import { css } from 'glamor';
import * as React from 'react';
import { Component } from 'react';
import {
  connect,
  Dispatch,
} from 'react-redux';
import {
  decrease,
  increase,
} from '../redux';

interface ICounterProps {
  value: number;
  dispatch: Dispatch<any>;
}
 
class Counter extends Component<ICounterProps, any> {
  render() {
    const {
      value,
      dispatch,
    } = this.props;
    return (
      <div>
        <Button type='default' text='decrease' onClick={() => dispatch(decrease())}/>
        <span {...css({ fontSize: '2em', display: 'inline-block', margin: '0 20px' })}>{value}</span>
        <Button type='primary' text='increase' onClick={() => dispatch(increase())}/>
      </div>
    );
  }
}
 
export default connect<any, any, any>(
  (state: any) => {
    return {
      value: state.counter.count,
    };
  },
)(Counter);
