type TChild = (innerProps: { images: string }) => JSX.Element | null;
 
const WithCarDetails = ({ children, carId }: { children: TChild, carId: string }) => {
  return <WithEntities
    groupName='carDetails'
    actionCreator={() => fetchListingsByIds([carId])}
    reducer={(_: any, action: Action<any>) => {
      return {
        images: get(action.payload.data, '[0].summary.images'),
      };
    }}
  >
    {children}
  </WithEntities>;
};
