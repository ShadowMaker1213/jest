import { css } from 'glamor';
// import { Dictionary } from '@types/lodash';
import map from 'lodash/map';
 
import * as React from 'react';
import { Component } from 'react';

// tslint:disable-next-line:interface-name
interface Dictionary<T> {
  [index: string]: T;
}
 
interface ICarInfoProps {
  carInfoLabels: Dictionary<string | number>;
  carInfoValues: Dictionary<string | number>;
}
 
export class CarInfo extends Component<ICarInfoProps, any> {
  render() {
    const { carInfoLabels, carInfoValues } = this.props;
    return (
      <ul {...css({ margin: 0, padding: 0 })}>
        {
          map(Object.keys(carInfoLabels), (labelKey: string, index: number) => {
            return (
              <li {...css({ display: 'flex', marginTop: '.3rem' })} key={index}>
                <span {...css(
                  { color: '#707070', width: '50%', marginRight: '1rem' })}>{carInfoLabels[labelKey]}:</span>
                <span {...css({ width: '50%' })}>{carInfoValues[labelKey]}</span>
              </li>
            );
          })
        }
      </ul>
    );
  }
}
