import { css } from 'glamor';
import map from 'lodash/map';
import * as React from 'react';
import { Component } from 'react';
const features = [
  'SECTION',
  'Traction control',
  'Bluetooth',
  'board computer',
  'ESP',
  'Electric mirrors',
  'Electric windows',
  'Driver & passenger airbag',
  'speakerphone',
  'air conditioner',
  'fog lights',
  'Radio / CD',
  'spare',
  'Power steering',
  'Servotronic',
  'daytime running lights',
  'Zentralv. with remote',
];

const clearfix = css({
  display: 'table',
  clear: 'both',
});
 
const featureItem = css({
  float: 'left',
  width: '33.33%',
  listStyleType: 'none',
  marginTop: '.3em',
  padding: 0,
});
 
const dot = css({
  background: '#7F7F7F',
  content: '',
  display: 'inline-block',
  height: '5px',
  width: '5px',
  borderRadius: '1px',
  verticalAlign: 'middle',
  marginRight: '.4em',
  position: 'relative',
  marginTop: '-.1em',
});
 
class Features extends Component<any, any> {
  render() {
    return (
      <section>
        <h2>Features</h2>
        <ul {...clearfix}>
          {
            map(features, (feature: string, index: number) => {
              return (
                <li {...featureItem} key={index}>
                  <span {...dot}></span>
                  <span>{feature}</span>
                </li>
              );
            })
          }
        </ul>
      </section>
    );
  }
}
 
export default Features;
