import { createRequestAction } from 'core-modules/request';
import * as queryString from 'query-string';

export const fetchListingsByIds = createRequestAction<string[]>('FETCH_LISTINGS_BY_IDS', (listingIds: string[]) => {
  return {
    method: 'GET',
    url: `/listings?${queryString.stringify({ listingIds: listingIds.join(',') })}`,
  };
});
